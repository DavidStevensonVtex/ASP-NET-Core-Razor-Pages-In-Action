@* Listing 3.8 Using the text tag to render multiline literal strings in the Welcome.cshtml file in the WebApplication1/Pages folder *@

@page
@model WebApplication1.Pages.WelcomeModel

@{
	ViewData["Title"] = "Welcome";
}

@functions {
	public class City
	{
		public string Name { get; set; }
		public string Country { get; set; }
	}
	List<City> cities = new List<City> {
		new City { Name = "London", Country = "UK" },
		new City { Name = "Paris", Country = "France" },
		new City { Name = "Rome", Country = "Italy" },
		new City { Name = "Berlin", Country = "Germany" },
		new City { Name = "Washington DC", Country = "USA" }
	};
}

<ul>
	@foreach (var city in cities)
	{
		if (city.Country == "UK")
		{
			<text>
				Country: @city.Country <br />
				Name: @city.Name
			</text>
		}
	}
</ul>
